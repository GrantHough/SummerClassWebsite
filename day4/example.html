Values of Input Elements

Try changing the different inputs. Then, press their associated button and check the console to see how the value changes.

Here is an example of using a span element, which displays inline, to change text styling inside a paragraph.

Now, I'm on a new paragraph element. It displays underneath the first paragraph because the paragraph tag displays block by default. And because it displays block, I can set a width of 50% and a margin of 50px

Below is the CSS, and JavaScript for this webpage

For HTML, just inspect element.

CSS

        body {
            display:flex;
            flex-direction: column;
            align-items:flex-start;
            padding: 0px 20px;
        }
        
        input {
            margin: 40 0px;
        }
        
        button {
            margin-top: 5px;
            margin-left: 20px;
        }
        
        hr {
            outline: 0px black solid;
            width: 100%;
            margin: 20px 0;
        }
        
        span {
            color: blue;
        }
        
        #second-paragraph {
            margin: 50px;
        }

JavaScript

        let checkbox = document.querySelector("#checkbox");
        let date = document.querySelector("#date");
        let file = document.querySelector("#file");
        let range = document.querySelector("#range");
        let color = document.querySelector("#color");

        let checkboxButton = document.querySelector("#checkbox-button");
        let dateButton = document.querySelector("#date-button");
        let fileButton = document.querySelector("#file-button");
        let rangeButton = document.querySelector("#range-button");
        let colorButton = document.querySelector("#color-button");

        let disappear = document.querySelector("#disappear");
        let reappear = document.querySelector("#reappear");
        let text = document.querySelector("#text");

        // These use arrow functions, they're just a shorter way of declaring a function

        checkboxButton.addEventListener("click", () => {
            console.log(checkbox.checked);
        });

        dateButton.addEventListener("click", () => {
            console.log(date.value);
        });

        fileButton.addEventListener("click", () => {
            console.log(file.value);
        });

        rangeButton.addEventListener("click", () => {
            console.log(range.value);
        });

        colorButton.addEventListener("click", () => {
            console.log(color.value);
        });


        disappear.addEventListener("click", function() {
            text.style.display = "none";
        })

        reappear.addEventListener("click", function() {
            text.style.display = "block";
        })
